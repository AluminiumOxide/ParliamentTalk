1 var mongoose = require('mongoose');
  2 var when = require('when');
  3 var User = mongoose.model('User');
  4 
  5 /**
  6  * @class
  7  * A controller to manage accounts
  8  */
  9 var Account = module.exports = {};
 10 
 11 /**
 12  * Signs up a new user
 13  * @param {http.request} req - The request object, with json body: { 
 14 				name: string, 
 15 				email: string, 
 16 				password: string }
 17  * @param {http.response} res - The response object:
 18 				<li> 400 if invalid data, with json body: {error: string, field: string} </li>
 19 				<li> 201 on success, with json body: {} </li>
 20  */
 21 Account.signUp = function(req, res) {
 22 	var account = new User();
 23 
 24 	// check and set the name
 25 	return when(account.setName(req.body.name))
 26 		.then(success => {
 27 			if(!success) {
 28 				return res.status(400).json({
 29 					error: 'Invalid user name', 
 30 					field: 'name'
 31 				});
 32 			}
 33 
 34 			// check and set the email
 35 			return when(account.setEmail(req.body.email))
 36 				.then(success => {
 37 					if(!success) {
 38 						return res.status(400).json({
 39 							error: 'Invalid email', 
 40 							field: 'email'
 41 						});
 42 					}
 43 
 44 					// check and set the password
 45 					return when(account.setPassword(req.body.password))
 46 						.then(success => {
 47 							if(!success) {
 48 								return res.status(400).json({
 49 									error: 'Invalid password', 
 50 									field: 'password'
 51 								});
 52 							}
 53 
 54 							// save the new user
 55 							return when(account.save())
 56 								.then(() => {
 57 									return res.status(201).json({});
 58 								});
 59 						});
 60 				});
 61 		});
 62 };
 63 
 64 /**
 65  * Signs a user in
 66  * @param {http.request} req - The request object
 67  * @param {http.response} res - The response object
 68  */
 69 Account.signIn = function(req, res) {
 70 	res.status(201).json({status: 'success'});
 71 };
 72 
 73 /**
 74  * Signs a user out
 75  * @param {http.request} req - The request object
 76  * @param {http.response} res - The response object
 77  */
 78 Account.signOut = function(req, res) {
 79 	res.status(201).json({status: 'success'});
 80 };
